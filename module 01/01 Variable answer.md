

### Решение Вариант 1 (Системный администратор)
**Файл:** `lab_vars_variant1.sh`

```bash
#!/usr/bin/env bash

# Лабораторная работа №1, Вариант 1
# Тема: Переменные, подстановки, массивы. Без циклов и условий.

# 1. Принимаем путь и расширение как аргументы командной строки
TARGET_DIR="$1"
EXTENSION="$2"

# 2. Строковая переменная RESULT — просто хранит команду (выполнять её пока не будем)
RESULT="ls -l"

# 3. Целочисленная переменная COUNT. 
# Считаем количество файлов, заканчивающихся на переданное расширение.
# Обратите внимание: ls -1 выводит по одному файлу в строке, 
# а подстановка $(...) превращает вывод в массив слов, затем считаем длину массива через #.
FILES_LIST=( $(ls -1 "$TARGET_DIR"/*"$EXTENSION" 2>/dev/null) )
COUNT=${#FILES_LIST[@]}

# 4. Массив FIRST_FIVE содержит первые 5 элементов из списка файлов.
# Используем срезы массивов: ${массив[@]:0:5} — взять с 0-го индекса 5 элементов.
FIRST_FIVE=( "${FILES_LIST[@]:0:5}" )

# 5. Формируем итоговый REPORT. 
# Обратите внимание на подстановку ${#COUNT} — но COUNT и так число.
# Используем элементы массива FIRST_FIVE по индексам 0 и 1.
REPORT="В папке $TARGET_DIR найдено $COUNT файлов с расширением $EXTENSION. Первые два: ${FIRST_FIVE[0]}, ${FIRST_FIVE[1]}"

# 6. Выводим отчет
echo "$REPORT"
```

### Решение Вариант 2 (Разработчик)
**Файл:** `lab_vars_variant2.sh`

```bash
#!/usr/bin/env bash

# Лабораторная работа №1, Вариант 2
# Тема: Ассоциативные массивы, арифметика, подстановки.

# 1. Объявляем ассоциативный массив CONFIG
declare -A CONFIG

# 2. Заполняем ключи системными значениями
CONFIG[user]="$USER"
CONFIG[home]="$HOME"
CONFIG[shell]="$SHELL"
# GROUPS — это массив. Берем первый элемент (индекс 0) через подстановку
CONFIG[group]="${GROUPS[0]}"

# 3. Создаем целочисленную переменную UID_NUM. 
# Для явного указания целого типа используем declare -i
declare -i UID_NUM
UID_NUM=$(( UID + 100 ))  # Арифметическая подстановка

# 4. Создаем многострочную переменную SUMMARY.
# В bash многострочность делается через переносы внутри двойных кавычек.
SUMMARY="Текущая конфигурация пользователя:
Имя пользователя: ${CONFIG[user]}
Домашняя папка: ${CONFIG[home]}
Оболочка: ${CONFIG[shell]}
Группа: ${CONFIG[group]}
UID + 100: $UID_NUM"

# 5. Выводим SUMMARY
echo "$SUMMARY"

# Завершаем скрипт с кодом возврата, равным длине SUMMARY.
# ${#SUMMARY} — оператор подсчета длины строки.
exit ${#SUMMARY}
```

### Решение Вариант 3 (Аналитик данных)
**Файл:** `lab_vars_variant3.sh`

```bash
#!/usr/bin/env bash

# Лабораторная работа №1, Вариант 3
# Тема: Строковые операции, массивы. Без sed/awk/wc/tr.

# 1. Читаем строку от пользователя
read -p "Введите произвольный текст: " INPUT_TEXT

# 2. Сохраняем в переменные:
LEN=${#INPUT_TEXT}                  # Длина строки
FIRST=${INPUT_TEXT:0:1}             # Извлечение подстроки с 0 по 1 символ
LAST=${INPUT_TEXT: -1}              # Обратите внимание на пробел перед -1! Извлекает последний символ.
CLEAN=${INPUT_TEXT,,}               # Преобразование в нижний регистр (Bash 4.0+)

# 3. Разбиваем текст по пробелам в индексированный массив WORDS.
# Важно: если не обернуть в кавычки, Bash разобьет по IFS (пробелы, табы).
# Используем () для создания массива из строки.
WORDS=( $INPUT_TEXT )

# 4. Собираем статистику. Количество слов — длина массива.
STATS="Количество слов: ${#WORDS[@]}, Первый символ: '$FIRST', Последний символ: '$LAST'"

# 5. Вывод
echo "$STATS"
echo "Текст в нижнем регистре: $CLEAN"
```

### Решение Вариант 4 (DevOps)
**Файл:** `lab_vars_variant4.sh`

```bash
#!/usr/bin/env bash

# Лабораторная работа №1, Вариант 4
# Тема: Ассоциативные и индексированные массивы, замена подстрок.

# 1. Объявляем ассоциативный массив SERVICES
declare -A SERVICES
SERVICES[web]="nginx"
SERVICES[db]="postgresql"
SERVICES[cache]="redis"

# 2. Создаем индексированный массив RUNNING, в который кладем только значения (без ключей).
# Для этого берем все значения ассоциативного массива через [@].
RUNNING=( "${SERVICES[@]}" )

# 3. Переменная COUNTER — длина массива RUNNING.
COUNTER=${#RUNNING[@]}

# 4. Формируем DEPLOY_CMD. Используем [*] для объединения всех элементов через первый символ IFS (пробел).
DEPLOY_CMD="systemctl restart ${RUNNING[*]}"

# 5. BACKUP_CMD — замена подстроки. 
# В ${var/что_меняем/на_что_меняем} меняем первое вхождение "restart" на "status".
BACKUP_CMD="${DEPLOY_CMD/restart/status}"

# 6. Вывод (две отдельные строки согласно ТЗ)
echo "$DEPLOY_CMD"
echo "$BACKUP_CMD"
```

### Решение Вариант 5 (Сетевой инженер)
**Файл:** `lab_vars_variant5.sh`

```bash
#!/usr/bin/env bash

# Лабораторная работа №1, Вариант 5
# Тема: Работа с позиционными параметрами, извлечение частей строк (IP).

# 1. Принимаем IP и порт как аргументы
IP_ADDR="$1"
PORT_INPUT="$2"

# 2. Переменная PROTOCOL
PROTOCOL="http"

# 3. Целочисленный PORT_CHECK.
# Используем арифметику с тернарным оператором? Нет, нельзя if.
# Поэтому используем подстановку параметров: если PORT_INPUT пуст — подставляем 80.
# ${VAR:-default} — если VAR не задана или пуста, используется "default".
declare -i PORT_CHECK
PORT_CHECK="${PORT_INPUT:-80}"

# 4. Массив URL_PARTS (можно создать, но по ТЗ он нужен).
# Массив объявляется просто скобками.
URL_PARTS=( "$PROTOCOL" "$IP_ADDR" "$PORT_CHECK" )

# 5. Собираем FULL_URL
FULL_URL="$PROTOCOL://$IP_ADDR:$PORT_CHECK"

# 6. Извлекаем первый октет IP-адреса.
# ${var%%.*} — удаляет всё, начиная с первой точки и справа (жадное удаление суффикса).
FIRST_OCTET="${IP_ADDR%%.*}"

# Формируем команду ping
PING_CMD="ping -c 1 $FIRST_OCTET"

# 7. Вывод
echo "$FULL_URL"
echo "$PING_CMD"
```
